Ext.define('PTS.model.CostCodeInvoice', {
    extend: 'Ext.data.Model',
    fields: [
        {
            name: 'costcodeinvoiceid',
            type: 'int',
            useNull: true
        },
        {
            name: 'costcodeid',
            type: 'int',
            useNull: true
        },
        {
            name: 'invoiceid',
            type: 'int',
            useNull: true
        },
        {
            name: 'datecharged',
            type: 'mydate',
            useNull: true
        },
        {
            name: 'amount',
            type: 'number',
            useNull: true
        }
    ],
    idProperty: 'costcodeinvoiceid',

    validations: [{
        type: 'length',
        field: 'costcodeid',
        min: 1
    }, {
        type: 'length',

        min: 1
    }, {
        type: 'presence',
        field: 'datecharged'
    }, {
        type: 'presence',
        field: 'amount'
    }],

    proxy: {
        type: 'rest',
        url : '../costcodeinvoice',
        reader: {
            type: 'json',
            root: 'data'
        }
    }
});
