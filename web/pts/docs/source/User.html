/**
 * File: app/model/User.js
 * Description: User model.
 */

Ext.define('PTS.model.User', {
    extend: 'Ext.data.Model',
    fields: [
        {
            name: 'loginid',
            mapping: 'loginid',
            type: 'int',
            persist: false
        },
        {
            name: 'contactid',
            mapping: 'contactid',
            type: 'int',
            persist: false
        },
        {
            name: 'groupid',
            mapping: 'groupid',
            type: 'int',
            persist: false
        },
        {
            name: 'username'
        },
        {
            name: 'firstname'
        },
        {
            name: 'lastname'
        },
        {
            name: 'middlename'
        },
        {
            name: 'suffix'
        },
        {
            name: 'groupname'
        },
        {
            name: 'acronym'
        }
    ],
    idProperty: 'loginid',

    proxy: {
        type: 'rest',
        url : '../userinfo',
        reader: {
            type: 'json',
            root: 'data'
        }
    },

    getUserName: function() {
        var first = this.get('firstname'),
            last = this.get('lastname');

        return last + ', ' + first;
    }
});
